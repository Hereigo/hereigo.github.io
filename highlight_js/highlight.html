            public class YourController : Controller
            {
                private readonly IWebHostEnvironment _webHostEnvironment;
        
                public YourController(IWebHostEnvironment webHostEnvironment)
                {
                    _webHostEnvironment = webHostEnvironment;
                }
        
                public ActionResult CheckFile()
                {
                    string webRootPath = _webHostEnvironment.WebRootPath;
                    string filePath = Path.Combine(webRootPath, "YourDirectory", "YourFile.txt");
        
                    if (System.IO.File.Exists(filePath))
                    {
                        // File exists
                        ViewBag.FileExists = true;
                    }
                    else
                    {
                        // File does not exist
                        ViewBag.FileExists = false;
                    }
        
                    return View();
                }
            }
